// Start of BODY

        // 1. Parse Deliveries safely
        class Delivery {
            String id; double x, y, weight, deadline; boolean delivered = false;
        }
        List<Delivery> dels = new ArrayList<>();
        for (JsonElement el : deliveries) {
            JsonObject d = el.getAsJsonObject();
            Delivery del = new Delivery();
            del.id = d.get("id").getAsString();
            del.x = d.get("x").getAsDouble();
            del.y = d.get("y").getAsDouble();
            del.weight = d.get("weight").getAsDouble();
            del.deadline = d.get("deadline").getAsDouble();
            dels.add(del);
        }
        
        // Greedily sort by deadline to maximize on-time successful deliveries (+100 points each)
        dels.sort(Comparator.comparingDouble(a -> a.deadline));

        // 2. Parse Drones
        class Drone {
            String id; double maxPayload; double x, y, time, battery;
            JsonArray path = new JsonArray();
        }
        List<Drone> droneList = new ArrayList<>();
        for (JsonElement el : drones) {
            JsonObject d = el.getAsJsonObject();
            Drone dr = new Drone();
            dr.id = d.get("id").getAsString();
            dr.maxPayload = d.get("max_payload").getAsDouble();
            dr.x = warehouseX; dr.y = warehouseY; dr.time = 0.0; dr.battery = 500.0;
            droneList.add(dr);
        }

        // 3. Parse Charging Stations with Capacity Tracking
        class Station {
            double x, y; int slots;
            List<double[]> reservations = new ArrayList<>();
        }
        List<Station> stations = new ArrayList<>();
        if (chargingStations.size() > 0) {
            for (JsonElement el : chargingStations) {
                JsonObject c = el.getAsJsonObject();
                Station st = new Station();
                st.x = c.get("x").getAsDouble();
                st.y = c.get("y").getAsDouble();
                st.slots = c.has("slots") ? c.get("slots").getAsInt() : 1;
                stations.add(st);
            }
        }

        JsonArray flightManifest = new JsonArray();

        // 4. Ray-Casting & Path Safety Helper
        class Helper {
            // High-precision ray-casting to ensure no NFZ is clipped
            boolean isPathClear(double x1, double y1, double x2, double y2, double startT, JsonArray nfzs) {
                double dist = Math.hypot(x2 - x1, y2 - y1);
                int steps = Math.max(1, (int) Math.ceil(dist * 10.0)); // Check every 0.1 units
                
                for (int i = 0; i <= steps; i++) {
                    double fraction = (double) i / steps;
                    double cx = x1 + (x2 - x1) * fraction;
                    double cy = y1 + (y2 - y1) * fraction;
                    double ct = startT + dist * fraction;
                    
                    for (JsonElement el : nfzs) {
                        JsonObject nfz = el.getAsJsonObject();
                        double tStart = nfz.get("T_start").getAsDouble();
                        double tEnd = nfz.get("T_end").getAsDouble();
                        
                        if (ct >= tStart && ct <= tEnd) {
                            String shape = nfz.get("shape").getAsString();
                            if (shape.equals("circle")) {
                                JsonArray center = nfz.getAsJsonArray("center");
                                if (Math.hypot(cx - center.get(0).getAsDouble(), cy - center.get(1).getAsDouble()) <= nfz.get("radius").getAsDouble() + 0.0001)
                                    return false;
                            } else if (shape.equals("rectangle")) {
                                JsonArray corners = nfz.getAsJsonArray("corners");
                                double nx1 = corners.get(0).getAsJsonArray().get(0).getAsDouble();
                                double ny1 = corners.get(0).getAsJsonArray().get(1).getAsDouble();
                                double nx2 = corners.get(1).getAsJsonArray().get(0).getAsDouble();
                                double ny2 = corners.get(1).getAsJsonArray().get(1).getAsDouble();
                                if (cx >= Math.min(nx1, nx2)-0.0001 && cx <= Math.max(nx1, nx2)+0.0001 && cy >= Math.min(ny1, ny2)-0.0001 && cy <= Math.max(ny1, ny2)+0.0001)
                                    return false;
                            }
                        }
                    }
                }
                return true;
            }

            // Calculates the exact time a drone must idle before an NFZ expires
            double getSafeWaitTime(double x1, double y1, double x2, double y2, double startT, JsonArray nfzs) {
                double waitTime = 0.0;
                while (!isPathClear(x1, y1, x2, y2, startT + waitTime, nfzs)) {
                    double dist = Math.hypot(x2 - x1, y2 - y1);
                    int steps = Math.max(1, (int) Math.ceil(dist * 10.0));
                    boolean found = false;
                    
                    for (int i = 0; i <= steps; i++) {
                        double fraction = (double) i / steps;
                        double cx = x1 + (x2 - x1) * fraction;
                        double cy = y1 + (y2 - y1) * fraction;
                        double ct = startT + waitTime + dist * fraction;
                        
                        for (JsonElement el : nfzs) {
                            JsonObject nfz = el.getAsJsonObject();
                            if (ct >= nfz.get("T_start").getAsDouble() && ct <= nfz.get("T_end").getAsDouble()) {
                                String shape = nfz.get("shape").getAsString();
                                boolean inside = false;
                                if (shape.equals("circle")) {
                                    JsonArray center = nfz.getAsJsonArray("center");
                                    if (Math.hypot(cx - center.get(0).getAsDouble(), cy - center.get(1).getAsDouble()) <= nfz.get("radius").getAsDouble() + 0.0001) inside = true;
                                } else if (shape.equals("rectangle")) {
                                    JsonArray corners = nfz.getAsJsonArray("corners");
                                    double nx1 = corners.get(0).getAsJsonArray().get(0).getAsDouble(), ny1 = corners.get(0).getAsJsonArray().get(1).getAsDouble();
                                    double nx2 = corners.get(1).getAsJsonArray().get(0).getAsDouble(), ny2 = corners.get(1).getAsJsonArray().get(1).getAsDouble();
                                    if (cx >= Math.min(nx1, nx2)-0.0001 && cx <= Math.max(nx1, nx2)+0.0001 && cy >= Math.min(ny1, ny2)-0.0001 && cy <= Math.max(ny1, ny2)+0.0001) inside = true;
                                }
                                if (inside) {
                                    waitTime = Math.max(waitTime, (nfz.get("T_end").getAsDouble() - startT) + 0.001);
                                    found = true; break;
                                }
                            }
                        }
                        if (found) break;
                    }
                    if (!found) break; // Infinite loop safety break
                }
                return waitTime;
            }

            // Checks station capacity and returns the time charging can actually begin
            double getActualChargeStartTime(Station st, double tArr) {
                double actualStart = tArr;
                while (true) {
                    int overlapping = 0;
                    double nextEnd = Double.MAX_VALUE;
                    for (double[] res : st.reservations) {
                        if (actualStart >= res[0] && actualStart < res[1]) {
                            overlapping++;
                            nextEnd = Math.min(nextEnd, res[1]);
                        }
                    }
                    if (overlapping < st.slots) break;
                    actualStart = nextEnd; // Jump to when the next slot frees up
                }
                return actualStart;
            }

            JsonObject createAction(String type, double x, double y, double t) {
                JsonObject action = new JsonObject();
                action.addProperty("x", x); action.addProperty("y", y); 
                // Formatting time to prevent floating point anomalies in output
                action.addProperty("t", Math.round(t * 1000.0) / 1000.0); 
                action.addProperty("action", type);
                return action;
            }
        }
        
        Helper h = new Helper();

        // 5. The Routing Engine (1-by-1 Safe Allocation)
        for (Delivery del : dels) {
            if (del.delivered) continue;

            Drone bestDrone = null;
            double bestTime = Double.MAX_VALUE;
            double bestWaitW = 0, bestWaitD = 0, bestWaitStationQueue = 0, bestChargeDuration = 0, bestWaitC = 0;
            Station bestStation = null;
            int bestPathType = 0; // 1 = Direct Route, 2 = Route via Charging Station

            for (Drone d : droneList) {
                if (del.weight > d.maxPayload) continue;

                // Scenario A: Direct route (Warehouse -> Deliver -> Warehouse)
                double waitW = h.getSafeWaitTime(warehouseX, warehouseY, del.x, del.y, d.time, noFlyZones);
                double tPickup = d.time + waitW;
                double distWD = Math.hypot(del.x - warehouseX, del.y - warehouseY);
                double tDeliver = tPickup + distWD;

                if (tDeliver <= del.deadline) {
                    double waitD = h.getSafeWaitTime(del.x, del.y, warehouseX, warehouseY, tDeliver, noFlyZones);
                    double tReturn = tDeliver + waitD + distWD;
                    double energyReq = distWD * (1 + del.weight) + distWD * 1.0;

                    if (energyReq <= d.battery && tReturn < bestTime) {
                        bestTime = tReturn;
                        bestDrone = d;
                        bestPathType = 1;
                        bestWaitW = waitW;
                        bestWaitD = waitD;
                    }
                }

                // Scenario B: If direct is invalid/too tight, check routes via charging stations
                for (Station st : stations) {
                    double distDC = Math.hypot(st.x - del.x, st.y - del.y);
                    double energyToStation = distWD * (1 + del.weight) + distDC * 1.0;

                    if (energyToStation <= d.battery) {
                        double waitD_C = h.getSafeWaitTime(del.x, del.y, st.x, st.y, tDeliver, noFlyZones);
                        double tArrStation = tDeliver + waitD_C + distDC;
                        
                        double actualChargeStart = h.getActualChargeStartTime(st, tArrStation);
                        double queueWaitTime = actualChargeStart - tArrStation;
                        
                        double batAtStation = d.battery - energyToStation;
                        double distCW = Math.hypot(warehouseX - st.x, warehouseY - st.y);
                        double energyCW = distCW * 1.0;

                        if (energyCW <= 500) { 
                            double chargeNeeded = Math.max(0, energyCW - batAtStation);
                            double chargeTime = chargeNeeded / 2.0;
                            double tChargeDep = actualChargeStart + chargeTime;
                            
                            double waitC_W = h.getSafeWaitTime(st.x, st.y, warehouseX, warehouseY, tChargeDep, noFlyZones);
                            double tReturn2 = tChargeDep + waitC_W + distCW;

                            if ((bestPathType == 0 || tReturn2 < bestTime) && tDeliver <= del.deadline) {
                                bestTime = tReturn2;
                                bestDrone = d;
                                bestPathType = 2;
                                bestWaitW = waitW;
                                bestWaitD = waitD_C;
                                bestWaitStationQueue = queueWaitTime;
                                bestChargeDuration = chargeTime;
                                bestWaitC = waitC_W;
                                bestStation = st;
                            }
                        }
                    }
                }
            }

            // 6. Manifest Compilation (Write sequence safely to JSON)
            if (bestDrone != null) {
                double distWD = Math.hypot(del.x - warehouseX, del.y - warehouseY);
                double tPickup = bestDrone.time + bestWaitW;
                
                if (bestWaitW > 0) bestDrone.path.add(h.createAction("WAIT", warehouseX, warehouseY, tPickup));
                
                JsonObject pickup = h.createAction("PICKUP", warehouseX, warehouseY, tPickup);
                JsonArray dids = new JsonArray(); dids.add(del.id);
                pickup.add("delivery_ids", dids);
                bestDrone.path.add(pickup);

                double tDeliver = tPickup + distWD;
                JsonObject deliver = h.createAction("DELIVER", del.x, del.y, tDeliver);
                deliver.addProperty("delivery_id", del.id);
                bestDrone.path.add(deliver);

                if (bestPathType == 1) { // Apply Direct Path
                    if (bestWaitD > 0) bestDrone.path.add(h.createAction("WAIT", del.x, del.y, tDeliver + bestWaitD));
                    double tReturn = tDeliver + bestWaitD + distWD;
                    bestDrone.path.add(h.createAction("RETURN", warehouseX, warehouseY, tReturn));
                    
                    bestDrone.time = tReturn;
                    bestDrone.battery = 500; // Auto-recharge at warehouse
                    del.delivered = true;
                    
                } else if (bestPathType == 2) { // Apply Path via Station
                    if (bestWaitD > 0) bestDrone.path.add(h.createAction("WAIT", del.x, del.y, tDeliver + bestWaitD));
                    double tArrStation = tDeliver + bestWaitD + Math.hypot(bestStation.x - del.x, bestStation.y - del.y);

                    bestDrone.path.add(h.createAction("CHARGE", bestStation.x, bestStation.y, tArrStation));
                    
                    double actualChargeStart = tArrStation + bestWaitStationQueue;
                    if (bestWaitStationQueue > 0) {
                        bestDrone.path.add(h.createAction("WAIT", bestStation.x, bestStation.y, actualChargeStart));
                    }
                    
                    double tChargeDep = actualChargeStart + bestChargeDuration;
                    bestDrone.path.add(h.createAction("CHARGE_COMPLETE", bestStation.x, bestStation.y, tChargeDep));
                    
                    // Lock the charging slot globally for other drones
                    bestStation.reservations.add(new double[]{actualChargeStart, tChargeDep});

                    if (bestWaitC > 0) bestDrone.path.add(h.createAction("WAIT", bestStation.x, bestStation.y, tChargeDep + bestWaitC));
                    
                    double distCW = Math.hypot(warehouseX - bestStation.x, warehouseY - bestStation.y);
                    double tReturn = tChargeDep + bestWaitC + distCW;
                    
                    bestDrone.path.add(h.createAction("RETURN", warehouseX, warehouseY, tReturn));
                    bestDrone.time = tReturn;
                    bestDrone.battery = 500; 
                    del.delivered = true;
                }
            }
        }

        // 7. Assemble the Final Valid Manifest Array
        for (Drone d : droneList) {
            if (d.path.size() > 0) {
                JsonObject manifestEntry = new JsonObject();
                manifestEntry.addProperty("drone_id", d.id);
                manifestEntry.add("path", d.path);
                flightManifest.add(manifestEntry);
            }
        }
